## 그래프 순회

### DFS

```c++
vector<int> adj[N];
bool visited[N];

void dfs(int s) {
    visited[s] = true;
    for (auto u : adj[s]) {
        if (!visited[u])
            dfs(u);
    }
}
```

### BFS

```c++
queue<int> q;
bool visited[N];
int distance[N];
// x에서 출발
visited[x] = true;
distance[x] = 0;
q.push(x);
while(!q.empty()) {
    int s = q.front(); q.pop();
    for (auto u : adj[s]) {
        if (visited[u]) continue;
        visited[u] = true;
        distance[u] = distance[s] + 1;
        q.push(u);
    }
}
```

### 벨만-포드 알고리즘

시작 노드에서 다른 모든 노드로 가는 최단 경로 구하기
음수 사이클 구하기 => n-1번의 라운드가 아닌 n번의 라운드를 하고 n번일때도 최단거리 갱신이 되면 존재하는 그래프

```c++
// Bellman-Ford Algorithm
vector<tuple<int, int, int>> edges;
int distance[n];

fill(distance, distance+n, INF);
distance[x] = 0;
for (int i = 0; i < n-1; i++) {
    for (auto e : edges) {
        int a, b, w;
        tie(a, b, w) = e;
        distance[b] = min(distance[b], distance[a]+w);
    }
}

// SPFA (더 빠른 버전)
vector<pair<int, int>> adj[n];
int distance[n];
queue<int> q;
bool inQueue[n];

fill(distance, distance+n, INF);
distance[x] = 0;
q.push(x);
inQueue[x] = true;
while(!q.empty()) {
    int u = q.front(); q.pop();
    inQueue[u] = false;
    for (pair<int, int> v : adj[u]) {
        int d, w;
        tie(d, w) = v;
        if (distance[u] + w < distance[d]) {
            distance[d] = distance[u] + w;
            if (!inQueue[d]) {
                q.push(d);
                inQueue[d] = true;
            }
        }
    }
}

// SPFA - SLF
vector<pair<int, int>> adj[n];
int distance[n];
deque<int> q;
bool inQueue[n];

fill(distance, distance+n, INF);
distance[x] = 0;
q.push_back(x);
inQueue[x] = true;
while(!q.empty()) {
    int u = q.front(); q.pop_front();
    inQueue[u] = false;
    for (pair<int, int> v : adj[u]) {
        int d, w;
        tie(d, w) = v;
        if (distance[u] + w < distance[d]) {
            distance[d] = distance[u] + w;
            if (!inQueue[d]) {
                inQueue[d] = true;
                if (distance[q.front()] > distance[d])
                    q.push_front(d);
                else
                    q.push_back(d);
            }
        }
    }
}
```

### 다익스트라 알고리즘

```c++
vector<pair<int, int>> adj[n];
priority_queue<pair<int, int>> q;
int distance[n];
bool processed[n];

fill(distance, distance+n, INF);
distance[x] = 0;
q.push({0, x});
while(!q.empty()) {
    int a = q.top().second; q.pop();
    if (processed[a]) continue;
    processed[a] = true;
    for (pair<int, int> u : adj[a]) {
        int b = u.first, w = u.second;
        if (distance[a] + w < distance[b]) {
            distance[b] = distance[a] + w;
            q.push({-distance[b], b});
        }
    }
}
```


```c++
const int INF = 1000000000;
vector<vector<pair<int, int>>> adj;

// d = 거리, p = parent
void dijkstra(int s, vector<int>& d, vector<int>& p) {
    int n = adj.size();
    d.assign(n, INF);
    p.assign(n, -1);
    vector<bool> u(n, false);
    
    d[s] = 0;
    for (int i = 0; i < n; i++) {
        int v = -1;
        for (int j = 0; j < n; j++) {
            if (!u[j] && (v == -1 || d[j] < d[v]))
                v = j;
        }
        
        if (d[v] == INF)
            break;
        
        u[v] = true;
        for (auto edge : adj[v]) {
            int to = edge.first;
            int len = edge.second;
            
            if (d[v] + len < d[to]) {
                d[to] = d[v] + len;
                p[to] = v;
            }
        }
    }
}
```

### Topological Sort (위상 정렬)

DFS 로 구하기 (cycle도 판별)

```c++
set<int> adj[1010];
bool visited[1010];
bool finished[1010];
bool cycle;
stack<int> topology;

void dfs(int s) {
    if (visited[s]) return;
    visited[s] = true;
    for (auto u : adj[s]) {
        if (!visited[u])
            dfs(u);
        else if (!finished[u])
           cycle = true;
    }
    finished[s] = true;
    topology.push(s);
}
```

indegree 를 이용한 방법 (indegree: 정점으로 들어오는 화살표)
이건 pq 반복문 후에 topology의 수가 실제 정점 수보다 작으면 싸이클이 있는 경우다.

```c++
vector<int> adj[1010];
int indegree[1010];
priority_queue<int> pq;
vector<int> topology;

// indegree가 0인 정점 pq에 넣기
for (int i = 1; i <= n; i++) {
    if (indegree[i] == 0)
        pq.push(-i);
}

while(!pq.empty()) {
    int here = -pq.top(); pq.pop();
    topology.push_back(here);
    for (int there : adj[here]) {
        indegree[there]--;
        if (indegree[there] == 0)
            pq.push(-there);
    }
}
```

### Minimum Spanning Tree (MST, 최소 스패닝 트리)

#### Kruskal's Algorithm (크루스칼 알고리즘)

```c++
vector<int> parent, depth;

void make_set(int v) {
    parent[v] = v;
    depth[v] = 0;
}

int find_set(int v) {
    if (v == parent[v])
        return v;
    return parent[v] = find_set(parent[v]);
}

void union_sets(int a, int b) {
    a = find_set(a);
    b = find_set(b);
    if (a != b) {
        if (depth[a] < depth[b])
            swap(a, b);
        parent[b] = a;
        if (depth[a] == depth[b])
            depth[a]++;
    }
}

struct Edge {
    int u, v, weight;
    bool operator<(Edge const& other) {
        return weight < other.weight;
    }
};

int n;
vector<Edge> edges;

int cost = 0;
vector<Edge> result;
parent.resize(n);
depth.resize(n);
for (int i = 0; i < n; i++)
    make_set(i);

sort(edges.init(), edges.end());

for (Edge e : edges) {
    if (find_set(e.u) != find_set(e.v)) {
        cost += e.weight;
        result.push_back(e);
        union_sets(e.u, e.v);
    }
}
```

### SCC (Strongly Connected Components, 강한 연결 요소)

#### Kosaraju's Algorithm (코사라주 알고리즘)

0. 원래 그래프, 역방향 그래프, 스택 준비
1. 모든 정점에서 dfs 돌아서 탐색 끝난 정점 stack에 넣기
2. stack에서 하나씩 빼서 역방향 그래프에서 dfs 돌리고 종료되면 그 그룹이 scc

```c++
const int MAX = 10000;
vector<vector<int>> scc;
vector<vector<int>> vt;
vector<vector<int>> rvt;
stack<int> st;
int visited[MAX + 1];

void dfs(int v) {
    visited[v] = true;
    for (int next : vt[v]) {
        if (visited[next])
            continue;
        dfs(next);
    }
    st.push(v);
}

void kosaraju(int v, int c) {
    visited[v] = true;
    scc[c].push_back(v);
    for (int next : rvt[v]) {
        if (visited[next])
            continue;
        kosaraju(next, c);
    }
}

int main() {
        // 모든 정점에서 dfs
    for (int i = 0; i < n; ++i) {
        if (visited[i])
            continue;
        dfs(i);
    }

    // visited 배열 초기화
    fill(visited, visited + n, 0);

    // stack에서 하나씩 빼서 역방향 dfs
    int group = 0;
    while(!st.empty()) {
        int cur = st.top();
        st.pop();
        if (visited[cur])
            continue;
        scc.emplace_back();
        kosaraju(cur, group++);
    }
}
```
